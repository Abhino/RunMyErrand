create table roomies(rm_id int(50) primary key,name varchar(50),contact int(20),email varchar(20),password varchar(20));
create table room_number(room_id int(50),rm_id int(50) ,foreign key (rm_id)  references  roomies (rm_id) );
create table tasks(task_id int(50),start_date datetime,end_date datetime,status varchar(20),points int(50),rm_id int(50), foreign key(rm_id) references roomies(rm_id));
create table relation(person_id int(50) primary key,task_id int(50),foreign key(task_id) references tasks(task_id));
